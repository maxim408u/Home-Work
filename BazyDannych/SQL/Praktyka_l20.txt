2
SELECT imie AS 'Imie_pracownika', nazwisko AS 'Nazwisko_Pracownika', stanowisko AS 'Zajmowane_Stanowisko'
FROM Pracownicy

3
SELECT DISTINCT stanowisko
FROM Pracownicy

4
SELECT id_projektu AS 'ID', nazwa_projektu AS 'Nazwa', data_rozpoczecia AS 'Start', data_zakonczenia AS 'Koniec', budzet AS 'Budzet_PLN' 
FROM Projekty

CZESC 2

5
SELECT imie, nazwisko
FROM Pracownicy
WHERE pensja >= 10000;

6
SELECT imie, nazwisko, data_zatrudnienia 
FROM Pracownicy
WHERE data_zatrudnienia LIKE '2022%';

7
SELECT imie, nazwisko, stanowisko
FROM Pracownicy
WHERE NOT stanowisko = 'Programista PHP';

8
SELECT imie, nazwisko, stanowisko
FROM Pracownicy
WHERE stanowisko = 'Project Manager' AND pensja > 12000;

9
SELECT nazwa_projektu, budzet
FROM Projekty
WHERE budzet < 100000 OR data_rozpoczecia LIKE '2023%';
